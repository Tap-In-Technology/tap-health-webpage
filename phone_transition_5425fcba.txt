                console.log('Phone header element NOT found!');
            }
            
            
            
            function handleScroll() {
                if (!phoneContainer || !landingSection) return;
                
                const rect = landingSection.getBoundingClientRect();
                const windowHeight = window.innerHeight;
                const scrollY = window.scrollY;
                
                // Calculate scroll progress (0 = top of page, 1 = bottom of page)
                const documentHeight = document.documentElement.scrollHeight;
                const scrollProgress = scrollY / (documentHeight - windowHeight);
                
                
                
                // Parallax effect for lightning background (only when section is in view)
                const lightningBg = document.querySelector('.lightning-bg-overlay');
                if (lightningBg && rect.top < windowHeight && rect.bottom > 0) {
                    const parallaxSpeed = 0.3; // Slower movement for parallax effect
                    const sectionOffset = rect.top;
                    const yPos = -(sectionOffset * parallaxSpeed);
                    lightningBg.style.transform = `translateY(${yPos}px)`;
                } else if (lightningBg) {
                    lightningBg.style.transform = 'translateY(0px)';
                }
                
                // Start phone transition at 96% scroll progress
                const transitionStart = 0.96;
                const transitionEnd = 1.0;
                
                if (scrollProgress >= transitionStart) {
                    // Calculate transition progress (0 to 1)
                    const transitionProgress = Math.min((scrollProgress - transitionStart) / (transitionEnd - transitionStart), 1);
                    
                    
                    // Fade in and slide up based on scroll progress
                    phoneContainer.style.opacity = transitionProgress;
                    // Start from bottom (top: 100%, translateY: 100px) and move to final position
                    const startY = 100; // Starting position from bottom
                    const endY = 25; // Ending position slightly higher than before
                    const currentY = startY + (endY - startY) * transitionProgress;
                    phoneContainer.style.transform = `translate(-50%, 0%) translateY(${currentY}vh)`;
                    phoneContainer.style.top = '100%'; // Always stay at bottom
                    
                    // Transition ready header out as phone transitions in
                    const readyHeader = document.querySelector('.ready-header');
                    if (readyHeader) {
                        readyHeader.style.opacity = (1 - transitionProgress).toString();
                    }
                    
                    // Show phone header when fully transitioned
                    if (transitionProgress >= 0.99) {
                        const phoneHeader = document.querySelector('.phone-header');
                        if (phoneHeader) {
                            phoneHeader.style.opacity = '1';
                            console.log('Header should now be visible - opacity set to 1');
                        } else {
                            console.log('Header element not found!');
                        }
                    } else {
                        const phoneHeader = document.querySelector('.phone-header');
                        if (phoneHeader) {
                            phoneHeader.style.opacity = '0';
                        }
                    }
                } else {
                    phoneContainer.style.opacity = '0';
                    phoneContainer.style.transform = 'translate(-50%, 0%) translateY(100px)';
                    phoneContainer.style.top = '100%';
                    
                    // Hide phone header and show ready header when phone is not transitioning
                    const phoneHeader = document.querySelector('.phone-header');
                    if (phoneHeader) {
                        phoneHeader.style.opacity = '0';
                    }
                    
                    const readyHeader = document.querySelector('.ready-header');
                    if (readyHeader) {
                        readyHeader.style.opacity = '1';
                    }
                }
            }
            
            window.addEventListener('scroll', handleScroll);
            handleScroll(); // Check initial state
        });
        
        // Scroll to top function
        function scrollToTop() {
            window.scrollTo({
                top: 0,
                behavior: 'smooth'
            });
        }
    </script>
